创建Node.js 应用

步骤一，引入require模块

				
	    var http = require("http");

步骤二，创建服务器

				
				//步骤一 引入require模块
var http = require("http");
//步骤二  创建服务器
http.createServer(function(req,response{
	//发送HTTP头部
	//HTTP状态值
	//内容类型：text/plain
	
	response.writeHead(200,"Content-Type","text/plain";
	//发送响应数据"Hello Wor！"
	response.end("< h1>Hello Wor!< /h1>");
}).listen(80);//端口
