JQ部分

< script src="libs/jquery-1.12.4.min.js">
< script type="text/javascript">
	$(function(){
		var i =0;
		var timer = null;  //声明空的变量
		fun();//回调函数
		function fun(){
			timer=setInterval(function(){
				if(i==4){ //判断图片放到最后一张  跳到第一张
					i=0;
		}
			//给图片添加淡入谈出效果
		$(".imgs li").eq(i).fadeIn().fadeTo().siblings().fadeOut();
		//让1,2,3,4样式跟随图片变化改变样式
	$(".ul li").eq(i).css("background", "aqua").siblings().css("background", "black");
			i++;
						
	},1500);}
			//鼠标移入事件
	$(".ul li").mouseenter(function(){  
	clearInterval(timer);//移入清除定时器
	$(".imgs li").eq($(this).index()).fadeIn().fadeTo().siblings().fadeOut();
		$(".ul li").eq($(this).index()).css("background", "aqua").siblings().css("background", "black");
		});
			//鼠标移出事件
		$(".ul li").mouseleave(function() {
		//调用函数
		fun();
		})
				
	})
		< / script>
